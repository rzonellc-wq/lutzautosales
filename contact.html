💡 Free Proxy Option: Deploy a free SMS proxy in 2 minutes using Render.com or Railway.app. After saving your Twilio keys, click "Get Proxy Code" below for a copy-paste Node.js server you can deploy for free.
📄 Free SMS Proxy Server (Node.js)

Deploy this to Render.com free tier — create a new Web Service, paste this code, set environment variables, and copy the URL into the Proxy field above.

const express = require('express');
const twilio = require('twilio');
const cors = require('cors');
const app = express();

app.use(cors());
app.use(express.json());

const client = twilio(
  process.env.TWILIO_SID,
  process.env.TWILIO_TOKEN
);

app.post('/send-sms', async (req, res) => {
  try {
    const { to, message } = req.body;
    await client.messages.create({
      body: message,
      from: process.env.TWILIO_FROM,
      to: to
    });
    res.json({ success: true });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

app.listen(3000, () => console.log('SMS proxy running'));

// package.json dependencies:
// "express": "^4.18.0"
// "twilio": "^4.0.0"
// "cors": "^2.8.5"

// Environment variables to set on Render:
// TWILIO_SID = your account SID
// TWILIO_TOKEN = your auth token
// TWILIO_FROM = your Twilio phone number
